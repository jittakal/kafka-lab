1. Get the application URL by running these commands:
{{- if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "kafeventstore.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "Metrics: http://$NODE_IP:$NODE_PORT/metrics"
  echo "Health: http://$NODE_IP:$NODE_PORT/health"
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  You can watch the status by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "kafeventstore.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "kafeventstore.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "Metrics: http://$SERVICE_IP:{{ .Values.service.metricsPort }}/metrics"
  echo "Health: http://$SERVICE_IP:{{ .Values.service.healthPort }}/health"
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "kafeventstore.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:9090/metrics to access metrics"
  echo "Visit http://127.0.0.1:8080/health to access health check"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 9090:{{ .Values.config.observability.metricsPort }} 8080:{{ .Values.config.observability.healthCheckPort }}
{{- end }}

2. Check deployment status:
  kubectl get deployment {{ include "kafeventstore.fullname" . }} -n {{ .Release.Namespace }}

3. Check pod logs:
  kubectl logs -f deployment/{{ include "kafeventstore.fullname" . }} -n {{ .Release.Namespace }}
